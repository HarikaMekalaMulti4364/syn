def dict_to_list_of_dicts(postprocessed_dict):
    postprocessed_results = []
    current_entry = {}
    
    for key, value in postprocessed_dict.items():
        # Assuming key contains some grouping information (e.g., 'group_key_field1', 'group_key_field2', ...)
        group_key = key.split('_')[0]  # Change this splitting logic based on how keys were created

        if group_key not in current_entry:
            if current_entry:  # Add the current entry if it's filled
                postprocessed_results.append(current_entry)
            current_entry = {}  # Start a new entry

        current_entry[key] = value

    if current_entry:  # Add the last entry
        postprocessed_results.append(current_entry)

    return postprocessed_results


# Example usage
postprocessed_results = dict_to_list_of_dicts(postprocessed_dict)
